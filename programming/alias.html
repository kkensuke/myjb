mkvjb() {
	mkdir -p ~/jupyter-book/$1
	cd ~/jupyter-book/$1
	python3 -m venv venv
	source venv/bin/activate
	/Users/Username/jupyter-book/$1/venv/bin/python3 -m pip install --upgrade pip
	pip install -U jupyter-book
	jb create $1
	cd $1
	jb build .
	open /Applications/Safari.app _build/html/index.html
}
